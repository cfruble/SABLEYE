dataSolver.decayPreProcessing

Script to process ENDF decay data into usable format.

This module reads ENDF decay data files, extracts half-lives, decay modes, and probabilities, and creates a JSON dictionary suitable for simulation models. It provides tools to parse file names, convert time units, and derive child isotopes according to decay chains.

Usage

Place ENDF decay files in './rawData/ENDF-B-VIII.0/decay/' Run the main script to generate './procData/decayData.json'.

Classes

decayProcessing Parse and process ENDF decay data files into a dictionary.

decayChain Loads decay dictionary and (partially) supports decay chain queries.

Functions

 65    @staticmethod
 66    def convert_to_seconds(time_string: str) -> float:
 67        """
 68        Convert various time unit strings to values in seconds.
 69
 70        Parameters
 71        ----------
 72        time_string : str
 73            Half-life string such as '1.23 D', '10.5 S', '4.7 Y', etc.
 74
 75        Returns
 76        -------
 77        float
 78            Equivalent seconds.
 79
 80        Raises
 81        ------
 82        ValueError
 83            If time unit is not recognized.
 84        """
 85        time_string = time_string.replace(">","")
 86        time_string = time_string.replace("<","")
 87
 88        # Conversion factors for different time units to seconds
 89        time_units_to_seconds = {
 90        'Y': 365 * 24 * 60 * 60,  # 1 year = 365 days = 24 hours/day = 60 minutes/hour = 60 seconds/minute
 91        'M': 60,   # 1 minute = 60 seconds
 92        'D': 24 * 60 * 60,        # 1 day = 24 hours/day = 60 minutes/hour = 60 seconds/minute
 93        'H': 60 * 60,             # 1 hour = 60 minutes * 60 seconds
 94        'S': 1,                   # 1 second = 1 second
 95        'MS': 1e-3,               # 1 millisecond = 1e-3 seconds
 96        'US': 1e-6,               # 1 microsecond = 1e-6 seconds
 97        'NS': 1e-9                # 1 nanosecond = 1e-9 seconds
 98        }
 99
100        # Split the string into two parts: value and unit
101        parts = time_string.strip().split()
102
103        # The first part is the significant value (can be a float or scientific notation)
104        significant_value = float(parts[0])
105
106        # The second part is the time unit
107        time_unit = parts[1].upper()  # Ensure it's uppercase for consistency
108
109        # Check if the time unit is in the dictionary
110        if time_unit in time_units_to_seconds:
111            # Convert the significant value to seconds
112            seconds = significant_value * time_units_to_seconds[time_unit]
113            return seconds
114        else:
115            raise ValueError(f"Unknown time unit: {time_unit}")

117    @staticmethod
118    def convert_fName_to_AAAZZZMMMM(fName:str) -> str:
119        """
120        Convert ENDF decay filename to canonical isotope code (AAAZZZMMMM). Example: ENDF decay file name in format of 'dec-092_U_235.endf' into '0922350000'
121
122        Parameters
123        ----------
124        fName : str
125            File name like 'dec-092_U_235.endf'
126
127        Returns
128        -------
129        str
130            Canonical isotope code like '0922350000'
131        """
132        metastable = 0
133        
134        name = fName[4:-5] # remove 'dec-' and '.endf'
135        if "m1" in name: # first metastable state
136            name = name.replace("m1","") # remvoe m1
137            metastable = 1
138        elif "m2" in name: #second metastable state
139            name = name.replace("m2","")
140            metastable = 2
141        name = re.sub(r'[^\d]', '', name) # remove nuclide name
142        # add back metastable state information
143        if metastable == 0:
144            name += "0000"
145        elif metastable == 1:
146            name += "0001"
147        elif metastable == 2:
148            name += "0010"
149        return name

152    @staticmethod
153    def childIsotopes(parent: str, decayModes: Union[List[str],None]) -> str:
154        """
155        Calculate child isotopes produced from various decay modes.
156
157        Parameters
158        ----------
159        parent : str
160            Canonical parent isotope code (AAAZZZMMMM).
161        decayModes : list of str or None
162            List of decay mode strings.
163
164        Returns
165        -------
166        list of str
167            Child isotope codes resulting from each decay mode.
168        """
169        if decayModes == None: # stable istotope case
170            return []
171        parentStr = str(parent)
172        Z = int(parentStr[0:3])
173        A = int(parentStr[3:6])
174        meta = parentStr[6:10]
175        childIsotopes = []
176        for decayMode in decayModes:
177            if decayMode == "B-": # beta -
178                Z += 1
179            elif decayMode == "A": # alpha
180                A -= 4
181                Z -= 2
182            elif decayMode == "EC" or decayMode == "B+": # electron capture
183                Z -= 1
184            elif decayMode == "EP" : # beta - proton decay
185                A -= 1
186            elif decayMode == "P": # proton release
187                A -= 1
188                Z -= 1
189            elif decayMode == "SF": # spontaneous fission
190                return ["SF"]
191            elif decayMode == "IT": # internal transition
192                if meta == "0001":
193                    meta = "0000"
194                elif meta == "0010":
195                    meta = "0001"
196                else:
197                    print(f"Error : metastable state {meta} not valid!")   
198            else: # decay mode not defined
199                raise ValueError
200                #print(f"Error : Decay mode {decayMode} not valid!")
201            childIsotopes.append(f"{Z:03}{A:03}{meta}")
202        return childIsotopes

204    def buildDecayDictionary(self, out_fName: str, out_fPath: str):
205        """
206        Build decay dictionary from ENDF files and save as JSON.
207
208        Reads each decay file, extracts parent, half-life, decay modes,
209        and branching probabilities, assembling them into a structured dictionary.
210
211        Output is saved as JSON in './procData/decayData.json'.
212
213        Parameters
214        ----------
215        out_fName : str
216            Output filename (not used, kept for compatibility).
217        out_fPath : str
218            Output path (not used in this implementation).
219
220        Returns
221        -------
222        None
223        """
224        # create empty lists to store data into
225        isotopes = []
226        halfLives = []
227        decayMode = []
228        decayProb = []
229
230        HLfails = 0
231        AssumedBeta = 0
232
233        # iterate through each file and extract: Parent Isotope, Half life [sec], Decay type
234        for fName in self.fNames:
235            # read in file as lines
236            fPath = os.path.join(self.decayENDF_fPath,fName)
237            with open(fPath,'r') as file:
238                lines = file.readlines()
239            
240            # search through lines for key phrases:
241            HL = False
242            DM = False
243            DM_temp = None
244            HL_temp = None
245            DP_temp = None
246            for line in lines:
247                # Check for other half life name and change
248                if "T1/2=" in line:
249                    line = line.replace("T1/2=","Parent half-life: ")
250                    #print(f"T1/2 replaced : {line}")
251
252                # check for stable case
253                if "Parent half-life:" in line and "STABLE" in line: # stable state
254                    DM = True
255                    DM_temp = None
256                    HL = True
257                    HL_temp = "Inf"
258                
259                elif "Parent half-life:" in line: # normal case
260                    HL = True
261                    HL_temp = line.strip("Parent half-life:")
262                    # remove last 8 characters from string
263                    HL_temp = HL_temp[:-10]
264                    HL_temp = HL_temp.strip()
265
266                    # convert string into halflife using function
267                    try:
268                        HL_temp = decayProcessing.convert_to_seconds(HL_temp)
269                    except:
270                        print(f"Time conversion fail for {fName} @ {HL_temp} ")
271                        HL_temp = None
272                        HLfails += 1
273
274                if "Decay Mode:" in line and DM == False: # found decay mode and NOT stable
275                    DM = True
276                    DM_temp = line.strip("Decay Mode:")
277                    # remove last 8 characters
278                    DM_temp = DM_temp[:-10]
279                    DM_temp = DM_temp.strip()
280                    DM_tempOG = DM_temp
281                    
282                    # remove double formating with no probabilities
283                    if DM_temp == "A, EC":
284                        DM_temp = "A"
285                    if DM_temp == "EC, A":
286                        DM_temp = 'EC'
287                    
288                    # add in formats for normal operations
289                    if (DM_temp == "B-") or (DM_temp == "A") or (DM_temp == "EC") or (DM_temp == "IT"):
290                        DM_temp = DM_temp.replace("B-","B-=100.00%")
291                        DM_temp = DM_temp.replace("A","A=100.00%")
292                        DM_temp = DM_temp.replace("EC","EC=100.00%")
293                        DM_temp = DM_temp.replace("IT","IT=100.00%")
294                    
295                    # standardize formating
296                    DM_temp = DM_temp.replace(" LE ","=")
297                    DM_temp = DM_temp.replace(" GT ","=")
298                    DM_temp = DM_temp.replace(" AP ","=")
299                    DM_temp = DM_temp.replace(" GE ","=")
300                    DM_temp = DM_temp.replace(" LT ","=")
301                    DM_temp = DM_temp.replace("<","=")
302                    DM_temp = DM_temp.replace(">","=")
303                    DM_temp = DM_temp.replace("?","")
304
305                    # split data into Decay Modes and Decay Probabilties
306                    # Regular expression to match the letter(s) and the percentage
307                    pattern = r'([A-Za-z\-]+)=(\d+\.\d+)%'
308
309                    # Find all matches using the regex pattern
310                    matches = re.findall(pattern, DM_temp)
311
312                    # Separate the matches into two lists
313                    DM_temp = [match[0] for match in matches]
314                    DP_temp = [float(match[1]) / 100 for match in matches]
315                    if DM_temp == [] and DP_temp == []: print(f"{DM_temp} {DP_temp} : \'{DM_tempOG}\'")
316                    
317
318                
319            if (HL and DM):
320                halfLives.append(HL_temp)
321                decayMode.append(DM_temp)
322                isotopes.append(decayProcessing.convert_fName_to_AAAZZZMMMM(fName))
323                decayProb.append(DP_temp)
324                #isotopes.append(fName[4:-5])
325            elif (HL is True and DM is False): # Fix broken cases
326                # Just assume that it is a B- decay since most errors are from here
327                # probably won't cause problems...
328                DM_temp = ["B-"] # Nothing has ever gone wrong making assumptions
329                DP_temp = [1.0]
330                halfLives.append(HL_temp)
331                decayMode.append(DM_temp)
332                isotopes.append(decayProcessing.convert_fName_to_AAAZZZMMMM(fName))
333                decayProb.append(DP_temp)
334                #isotopes.append(fName[4:-5])
335                if self.consoleLog : print(f"Assumed B- decay for {fName}")
336                AssumedBeta += 1
337            else:
338                if self.consoleLog : print(f"HL DM read fail for {fName} @ HL:{HL_temp} DM:{DM_temp}")
339        
340        
341        # create dictionary for output
342        decayDict = {}
343        dictFails  = 0
344        for i in range(len(isotopes)):
345            parent = isotopes[i]
346            if halfLives[i] == None : continue # catch error and remove bad data
347            try:
348                decayDict[parent] = {
349                    'decayConst' :  1/float(halfLives[i]),
350                    'childNames' : decayProcessing.childIsotopes(parent,decayMode[i]),
351                    'childProbs' : decayProb[i]
352                    }
353            except:
354                print(f"DictFail for : {halfLives[i]} {decayMode[i]} {decayProb[i]}")
355                dictFails += 1
356                continue
357            
358        
359        print("  ***** RUN SUMMARY *****")
360        print(f"HL DM read fails      : {len(self.fNames)-len(halfLives)}")
361        print(f"Time conversion fails : {HLfails}")
362        print(f"Assumed B- decays     : {AssumedBeta}")
363        print(f"Decay dictionary fails : {dictFails}")
364        
365        with open("./procData/decayData.json",'w') as file:
366            json.dump(decayDict, file, indent=4)
367
368        # create formatted output file
369        #linesOut = []
370        #with open("decayData.csv",'w') as file:
371        #    for i in range(len(halfLives)):
372        #        linesOut.append( f"{isotopes[i]},{halfLives[i]},{decayMode[i]}\n" )
373        #    # write output to file
374        #    file.writelines(linesOut)

 67    def convert_to_seconds(time_string: str) -> float:
 68        """
 69        Convert various time unit strings to values in seconds.
 70
 71        Parameters
 72        ----------
 73        time_string : str
 74            Half-life string such as '1.23 D', '10.5 S', '4.7 Y', etc.
 75
 76        Returns
 77        -------
 78        float
 79            Equivalent seconds.
 80
 81        Raises
 82        ------
 83        ValueError
 84            If time unit is not recognized.
 85        """
 86        time_string = time_string.replace(">","")
 87        time_string = time_string.replace("<","")
 88
 89        # Conversion factors for different time units to seconds
 90        time_units_to_seconds = {
 91        'Y': 365 * 24 * 60 * 60,  # 1 year = 365 days = 24 hours/day = 60 minutes/hour = 60 seconds/minute
 92        'M': 60,   # 1 minute = 60 seconds
 93        'D': 24 * 60 * 60,        # 1 day = 24 hours/day = 60 minutes/hour = 60 seconds/minute
 94        'H': 60 * 60,             # 1 hour = 60 minutes * 60 seconds
 95        'S': 1,                   # 1 second = 1 second
 96        'MS': 1e-3,               # 1 millisecond = 1e-3 seconds
 97        'US': 1e-6,               # 1 microsecond = 1e-6 seconds
 98        'NS': 1e-9                # 1 nanosecond = 1e-9 seconds
 99        }
100
101        # Split the string into two parts: value and unit
102        parts = time_string.strip().split()
103
104        # The first part is the significant value (can be a float or scientific notation)
105        significant_value = float(parts[0])
106
107        # The second part is the time unit
108        time_unit = parts[1].upper()  # Ensure it's uppercase for consistency
109
110        # Check if the time unit is in the dictionary
111        if time_unit in time_units_to_seconds:
112            # Convert the significant value to seconds
113            seconds = significant_value * time_units_to_seconds[time_unit]
114            return seconds
115        else:
116            raise ValueError(f"Unknown time unit: {time_unit}")
